    public SprogMessage frame() {
      int j = 2;
      SprogMessage f = new SprogMessage(MAXSIZE);
      f.setElement(0, STX);
      f.setElement(1, STX);
      for (int i = 0; i < _nDataChars; i++) {
        if (_dataChars[i] == STX
            || _dataChars[i] == ETX
            || _dataChars[i] == DLE) {
          f.setElement(j++, DLE);
        }
        f.setElement(j++, _dataChars[i]);
      }
      f.setElement(j++, ETX);
      f._nDataChars = j;
      return f;
    }
