	void setupMenu() {
		m_menu_bar = new JMenuBar();
		setJMenuBar(m_menu_bar);
		addMenu("File");
		addMenuItem(sReadSourceMapping);
		addMenuItem(sReadTargetMapping);
		addMenuSeparator();
		addMenuItem(sSaveSourceMapping);
		addMenuItem(sSaveTargetMapping);
		addMenuSeparator();
		addMenuItem(sSave);
		addMenuItem(sExit);
		addMenu("Mapping");
		addMenuItem(sAutoSrcMapping);
		addMenuItem(sAutoTrgMapping);
		addMenuSeparator();
		addMenuItem(sRmSrcMapping);
		addMenuItem(sRmTrgMapping);
		addMenuItem(sAutoTrgMapping);
		addMenuSeparator();
		addMenuItem(sRmSrcMapping);
		addMenuItem(sRmTrgMapping);
		addMenu("View");
		addMenuItem(sShowAllMapping);
		addMenuItem(sHideSourceMapping);
		addMenuItem(sHideTargetMapping);
		addMenuItem(sExpandAllTrees);
		JMenu search=addMenu("Search");
		addMenu("Start", search);
		m_search_nst=addMenuItem("New source tree");
		m_search_ost=addMenuItem("Old source tree");
		m_search_ott=addMenuItem("Old target tree");
		m_search_ntt=addMenuItem("New target tree");
		addMenu("Next", search);
		m_next_nst=addMenuItem("New source tree",false);
		m_next_ost=addMenuItem("Old source tree",false);
		m_next_ott=addMenuItem("Old target tree",false);
		m_next_ntt=addMenuItem("New target tree",false);
	}
